 12:      * 执行全局过滤器
 13:      * 1、首选如果当前请求的shell等于首选启动指令，则忽略过滤器
 14:      * 2、系统会根据配置的全局过滤器的数组下标递增执行，详见DIFilterMap::getGlobalFilters() @ filtermap.php
 15:      * 3、如果shell等于全局过滤器中配置的例外项，也忽略对应的过滤器
 16:      */
 17:     static function execGlobalFilter(){
 18:         $shell = DIRuntime::getItem('shell');
 19:         $isDefault = !strcasecmp($shell, DIUrlShell::$_default_shell);
 20:         if ($isDefault) {
 21:             return;//首选启动指令将忽略全局过滤器
 22:         }
 23:         
 24:         $list = DIFilterMap::getGlobalFilters();
 25:         $execlist = array();
 26:         foreach ($list as $k => &$l) {
 27:             if (!is_array($l)) continue;
 28:             $exclude = false;
 29:             foreach ($l as $ll) {
 30:                 if (DI_DO_MODULE == $ll) { $exclude = true; break; }
 31:                 if (DI_DO_MODULE .'/'. DI_DO_FUNC == $ll) { $exclude = true; break; }
 32:             }
 33:             if ($exclude) continue;
 34:             $execlist[] = $k;
 35:         }
 36:         self::invokeFilter($execlist);
 37:     }

 39:     /**
 40:      * 执行有特定作用点的过滤器
 41:      * @author biao
 42:      * @since 2014-10-28
 43:      * @param string $spot 过滤器作用点
 44:      * <pre>
 45:      *      格式：[Do名称/方法名] ，不含“[]”，大小写敏感
 46:      *      例如self::exec('test/lets');
 47:      * </pre>
 48:      */
 49:     static function execSpecialFilter($spot){
 50:         $validate = is_string($spot) && false !== strpos($spot, '/') && 2 == count($ex = explode('/', $spot));
 51:         if (!$validate) { throw new DIException('DIFilterUtil::exec() FILE ' . __FILE__ . ' LINE '. __LINE__ . ' $spot参数错误，格式必须为[Do名/方法]，当前序列化值：' . serialize($spot)); }
 52:         
 53:         $ns = DIFilterMap::getNeedsMap();
 54:         $ws = DIFilterMap::getWithoutMap();
 55: 
 56:         $lneed = self::loop($ns, $spot, 'getNeedsMap');//需要的过滤器
 57:         $lwithout = self::loop($ws, $spot, 'getWithoutMap');//不需要的过滤器
 58:         
 59:         //根据DIFilterMap::$need的默认值，来决定一类过滤器（同时配置了“需执行”和“不执行”的）的去留。最终可执行的过滤器保存在$needs中
 60:         $needlist = $lneed['list'];
 61:         $withoutlist = $lwithout['list'];
 62:         if (! DIFilterMap::$need) {
 63:             foreach ($withoutlist as &$w) {
 64:                 while (is_int($index = array_search($w, $needlist))) {
 65:                     array_unset($needlist, $index);
 66:                 }
 67:             }
 68:         }
 69:         $sortlist = self::setSort($needlist, $spot);
 70:         self::invokeFilter($sortlist);
 71:     }

 73:     /*
 74:      * 通过循环来验证$spot是否在$loop中配置，并获取$spot对应的过滤器名称集合
 75:      * 返回：flag 是否存在配置 list 过滤器集合
 76:      * flag下标其实没用，只是用来证明有没有配置过need/without规则，但不能证明具体对应了哪些规则
 77:      */
 78:     static private function loop($loop, $spot, $loopname){
 79:         if (!is_array($loop)) { throw new DIException("过滤规则[ {$loopname} ]配置格式错误"); }
 80:         
 81:         $list = array();
 82:         $flag = false;
 83:         foreach ($loop as $i => &$l) {
 84:             if (!is_array($l)) { throw new DIException("过滤规则[ {$loopname} - {$i} ]配置格式错误"); }
 85:             if (in_array($spot, $l)) { $flag = true; $list[] = $i; continue; }
 86: 
 87:             foreach ($l as $k => &$ll) {
 88:                 if (!is_string($ll)) { throw new DIException("过滤器作用点[ {$loopname} - {$i} - {$k} ]配置格式错误"); }
 89:                 $ex = explode('/', $spot);
 90:                 if ($ll == $ex[0]) { $flag = true; $list[] = $i; continue; }
 91:             }
 92:         }
 93:         
 94:         return array('flag' => $flag, 'list' => $list);
 95:     }
 96:     
 97:     /*
 98:      * 调整特定作用点过滤器的执行顺序
 99:      * 例如执行点spot的可执行过滤器有S1={A,B,C,D,E},其优先级配置为S2={C,B,F}，
100:      *     则最终排序结果为{C,B,A,D,E}。
101:      *     其中，S2的F被删除，S1的{A,D,E}按原顺序追加到S2尾部。
102:      *     集合运算步骤如下：
103:      *     (1)通过
104:      *     S1 ∩ S2 = {C,B} = N1
105:      *     或 S2 - (S2 - S1) = {C,B,F} - {F} = {C,B} = N1
106:      *     得到S1中已对号入座的队列N1；
107:      *     (2)取S1中剩余未对号入座的队列
108:      *     S1 - N1 = {A,D,E} = N2
109:      *     (3)将N2追加到N1尾部，得到最终结果R
110:      *     N1 <+ N2 = {C,B,A,D,E} = R
111:      * @author biao
112:      * @since 2014-11-26
113:      */
114:     static private function setSort($list, $spot) {
115:         $spSort = DIFilterMap::getSpecialFilterSort();
116:         $dfSort = DIFilterMap::getDefaultFilterSort();
117:         $sort = isset($spSort[$spot]) ? $spSort[$spot] : $dfSort;
118:         //按照顺序对号入座
119:         foreach ($sort as $i=>$s) {
120:             $index = array_search($s, $list);
121:             if (is_int($index)) {
122:                 array_unset($list, $index);
123:             } else {
124:                 array_unset($sort, $i);
125:             }
126:         }
127:         //将剩余未对号入座的追加到sortlist尾部
128:         foreach ($list as $l) {
129:             array_push($sort, $l);
130:         }
131:         return $sort;
132:     }
