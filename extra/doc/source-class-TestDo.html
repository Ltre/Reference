273:         $key_exist = !empty($key) || is_numeric($key);
274:         switch ($opt) {
275:             case 'get':
276:                 $key_exist || dump(session(), true);//--END
277:                 session_exists($key) ? exit("$key => " . session($key)) : exit('会话中不存在'.DI_SESSION_PREFIX.$key);//--END
278:             case 'set':
279:                 $key_exist || die('必须指定key');//--END
280:                 session($key, $value);
281:                 exit("[ $key => $value ]设置成功");//--END
282:             case 'remove':
283:                 $key_exist || die('必须指定key');
284:                 session_remove($key, true);
285:                 exit("会话[ $key ]删除成功");//--END
286:             case 'clear':
287:                 session(null);
288:                 exit('已清空本应用专用的会话');//--END
289:         }
