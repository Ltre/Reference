222:     if(! is_object($obj)){
223:         throw new DIException("{$obj}对象不存在或非对象类型");

347:     $num = count(func_get_args());
348:     $prefix = null===$prefix ? DI_SESSION_PREFIX : $prefix;

367:     }

458:  * 也可以强制用第二个参数将shell指定为文件路径
459:  * @notice

465: function redirect($shell, $params = array()){
466:     //如果第二参数设置为false，则将shell视为文件路径
467:     if (false === $params) {
468:         require $shell;
469:         exit;
470:     }
471:     
472:     $ar = explode('/', $shell);
473:     $validate = is_string($shell) && !empty($ar) && 2 == count($ar);
474:     if ($validate) {
475:         $do = $ar[0] . DI_KEYWORD_DO;
476:         $func = $ar[1];
477:         $d = component_exist($do);
478:         $m = method_exists($do, $func);
479:         if ($d && $m) {
480:             invoke_method(new $do(), $func, $params);
481:             exit;
482:         }
483:     }
484:     
485:     if (file_exists($shell)) {
486:         require $shell;
487:         exit;
488:     }
489:     
490:     throw new DIException("shell参数[{$shell}]不是一个有效的DO指令，也不是一个文件路径，无法跳转");
491: }
