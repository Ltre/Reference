160:     
161:     protected static $_conn;//与DIModel::$_conn存在引用关联，详见本类的&connect()
162:     protected $table;//当前表名
163:     protected $_M;//DIModel对象授予操作权限

166:         $this->table = $table;

171:      * 连接并获取句柄
172:      * @return PDO
173:      */
174:     function &connect(){
175:         $dsn = 'mysql:host=' . DIDBConfig::$host;
176:         $dsn .= ';dbname=' . DIDBConfig::$db;
177:         $dsn .= ';port=' . DIDBConfig::$port;
178:         $options = array(
179:             PDO::MYSQL_ATTR_INIT_COMMAND => 'set names utf8'
180:         );
181:         self::$_conn = new PDO($dsn, DIDBConfig::$user, DIDBConfig::$pwd, $options);
182:         return self::$_conn;

184:     
185:     /**
186:      * 示例
187:      * $month = date("Y-m-d", strtotime($date));
188:      * $sql = "SELECT * FROM pai2_signin WHERE uid =:uid AND signin_date LIKE '{$month}%' ORDER BY signin_id ASC ";
189:      * $rs = $this->query($sql, array('uid' => $uid));
190:      */
191:     function query($sql, $params=array()){
192:         $this->sql = $sql;
193:         $sth = $this->_bindParams($sql, $params);
194:         if($sth->execute()) return $sth->fetchAll(PDO::FETCH_OBJ);
195:         $err = $sth->errorInfo();
196:         throw new DIException('Database SQL: "' . $sql. '". ErrorInfo: '. $err[2], 1);
197:     }
198:     
199:     function insert($data = array()){
200:         foreach($data as $k=>$v){
201:             $keys[] = "`{$k}`";
202:             $prevalues[] = $k;
203:             $values[] = $this->escape($v);
204:         }
205:         $sql = "INSERT INTO " . $this->table . " (" . join(', ', $keys) . ") VALUES ( :" . join(', :', $prevalues) . ")";
206:         $sth = $this->_bindParams($sql, $data);
207:         if ($sth->execute()) return self::$_conn->lastInsertId();
208:         $err = $sth->errorInfo();
209:         throw new Exception('Database SQL: "' . $sql. '". ErrorInfo: '. $err[2], 1);
210:     }
211:     
212:     function update(array $cond, $data = array()){
213:         if( empty($cond) ) return false;
214:         foreach ($data as $k=>$v){
215:             $setstr[] = "`{$k}`=".$this->escape($v);
216:         }
217:         return $this->execute("UPDATE ".$this->table." SET ".implode(', ', $setstr).$this->_where($cond));
218:     }

223:      * 还没试过符号：*、/、%
224:      */
225:     function alter(array $cond, $field, $optval = '+1') {
226:         if( empty($cond) ) return false;

232:      * @param array $cond

271:     /**
272:      * 两种方式统计总数
273:      * 示例：$t = DIModelUtil::supertable('table');
274:      * 1、$t->count(array('id'=>2));
275:      * 2、$t->count(array('id > :id'), array('id'=>2));
276:      */
277:     function count($conditions = null, $bindparams = array()){
278:         $select = "SELECT COUNT(*) AS total FROM {$this->table} ";
279:         $needbind = false;//是否需要预处理参数
280:         $where = ' WHERE 1 ';
281:         foreach ($conditions as $i=>$c) {
282:             if (is_numeric($i)) {
283:                 $needbind = true;
284:                 $where .= " AND {$c} ";
285:             }
286:         }
287:         if (!$needbind) {
288:             $count = $this->query($select . $this->_where($conditions));
289:         } else {
290:             $count = $this->query("{$select} {$where}", $bindparams);
291:         }
292:         return isset($count[0]->total) && $count[0]->total ? $count[0]->total : 0;
293:     }

295:     function execute($sql=null, $params=array()){
296:         $this->sql = $sql;
297:         $sth = $this->_bindParams( $sql, $params );
298:         if( $sth->execute() ) return $sth->rowCount();
299:         $err = $sth->errorInfo();
300:         throw new Exception('Database SQL: "' . $sql. '". ErrorInfo: '. $err[2], 1);
301:     }

303:     private function _bindParams($sql, $params=array()){
304:         $sth = self::$_conn->prepare($sql);
305:         if(is_array($params) && !empty($params)){
306:             foreach($params as $k=>&$v){
307:                 $sth->bindParam($k, $v);
308:             }
309:         }
310:         return $sth;
311:     }

313:     private function _where($conditions = array()){
314:         if(is_array($conditions) && !empty($conditions)){
315:             $join = array();
316:             foreach( $conditions as $key => $condition ){
317:                 $condition = $this->escape($condition);
318:                 $join[] = "`{$key}` = {$condition}";
319:             }
320:             return " WHERE ".join(" AND ",$join);
321:         }
322:         return ' ';
323:     }

325:     private function escape($str){
326:         if(is_null($str))return 'null';
327:         if(is_bool($str))return $str ? 1 : 0;
328:         if(is_int($str))return (int)$str;
329:         if(@get_magic_quotes_gpc())$str = stripslashes($str);
330:         return self::$_conn->quote($str);
331:     }

333:     //参数：当前页码，页内条数上限，可见页码数，记录总条数
334:     function pager($page, $pageSize = 10, $scope = 10, $total)
335:     {
336:         $total_page = ceil( $total / $pageSize );
337:         $page = max(intval($page), 1);
338:         $this->_M->page = array(
339:             'total_count' => $total,
340:             'page_size'   => $pageSize,
341:             'total_page'  => $total_page,
342:             'first_page'  => 1,
343:             'prev_page'   => ( ( 1 == $page ) ? 1 : ($page - 1) ),
344:             'next_page'   => ( ( $page == $total_page ) ? $total_page : ($page + 1)),
345:             'last_page'   => $total_page,
346:             'current_page'=> $page,
347:             'all_pages'   => array(),
348:             'offset'      => ($page - 1) * $pageSize,
349:             'limit'       => $pageSize,
350:         );
351:         $scope = (int)$scope;
352:     
353:         if($total <= $pageSize){
354:             $this->_M->page['all_pages'] =  array();
355:         }elseif($total_page <= $scope ){
356:             $this->_M->page['all_pages'] = range(1, $total_page);
357:         }elseif( $page <= $scope/2) {
358:             $this->_M->page['all_pages'] = range(1, $scope);
359:         }elseif( $page <= $total_page - $scope/2 ){
360:             $right = $page + (int)($scope/2);
361:             $this->_M->page['all_pages'] = range($right-$scope+1, $right);
362:         }else{
363:             $this->_M->page['all_pages'] = range($total_page-$scope+1, $total_page);
364:         }
365:         return $this->_M->page;
366:     }
